class Solution {
    public static int longestSubseq(int n, int[] a) {
        // code here
         int maxi = 1;
        int f[] = new int[1002];
        for(int i:a){
            f[i] = 1 + Math.max(f[i-1], f[i+1]);
            maxi = Math.max(maxi, f[i]);
        }
        return maxi;
    }
