3&nbsp;&nbsp; &nbsp;0<br>
4&nbsp;&nbsp; &nbsp;2<br>
5&nbsp; &nbsp; 0<br>
6&nbsp; &nbsp; 1<br>
7&nbsp; &nbsp; 0<br>
8&nbsp; &nbsp; 3<br>

</span>
</pre>
